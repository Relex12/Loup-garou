// déroulement de la partie


premiere_nuit ()
tant que (! quelquun_gagne())
{
	liste_morts = deroulement_nuit ()
	pour (joueur in liste_morts)
		action_mort (joueur)
	si (quelquun_gagne())
		sortir
	liste_morts = deroulement_jour ()
	pour (joueur in liste_morts)
		action_mort (joueur)
}
